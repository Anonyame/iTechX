\begin{homeworkProblem}
Let $\SS=\{\st0,\st1\}$ and let $C_2 = \set{ztz}z\in\st0^*$
and $t\in\st0^*\st1\st0^*\st1\st0^*$, where $|t|=|z|\setend$.
\begin{enumerate}[(a)]
	\item Show that $C_2$ is not a CFL.
	\item Is $C_2 \union (\SS\SS\SS)^*$ a CFL?  Why or why not?
	\item Is $C_2 \union \SS(\SS\SS\SS)^*$ a CFL?  Why or why not?
\end{enumerate}
\end{homeworkProblem}


\begin{homeworkProblem}
Let $G = (V, \Sigma, R, \tTk{stmt})$ be the following grammar.
$ \Sigma = \{\st{if}, \st{condition}, \st{then}, \st{else},
\st{a:=1}\},\\
V = \{\tTk{stmt}, \tTk{if-then}, \tTk{if-then-else}, \tTk{assign}\} $ 
and the rules are:
\begin{grammar}
	<\textsc{stmt}> $\cfgarrow$ $\tTk{assign} \cfgor \tTk{if-then} \cfgor \tTk{if-then-else}$ 
	
	<\textsc{if-then}> $\cfgarrow$ $\st{if condition then}\; \tTk{stmt} $ 
	
	<\textsc{if-then-else}> $\cfgarrow$ $\st{if condition then}\; \tTk{stmt} \; \st{else}\; \tTk{stmt} $ 
	
	<\textsc{assign}> $\cfgarrow$ $\st{a:=1}$ 

\end{grammar}
\begin{enumerate}[(a)]
	\item Show that $G$ is ambiguous.
	\item Give a new unambiguous grammar that generates $L(G)$. \\
	(You do not need to prove that your grammar works or that it is
	unambiguous, but please add a few comments about why it does work
	to help the grader.)
\end{enumerate} 
\end{homeworkProblem}

\begin{homeworkProblem}
Let the \textbf{\emph{rotational closure}} of language $A$
be $\text{RC}(A)=\{ yx\mid xy\in A$ where $x,y\in\Sigma^\ast\}$.
Show that the class of \textsf{CFL}s is closed under rotational closure.
\end{homeworkProblem}
