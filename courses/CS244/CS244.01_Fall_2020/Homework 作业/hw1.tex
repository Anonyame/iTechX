Let $\Sigma = \{\mathsf{0}, \mathsf{1}\}$ if not otherwise specified.

\begin{homeworkProblem}
Let $A$ be the languages that every 1 has at least 2 zeros following immediately after. Show that $A$ is regular in the following ways:
\begin{enumerate}[(a)]
    \item by giving an $\mathsf{NFA}$ that recognizes $A$,
    \item by giving a $\mathsf{DFA}$ that recognizes $A$,
    \item by giving a regular expression that describes $A$, and
    \item by giving a right linear grammar that describes $A$.
\end{enumerate}
\end{homeworkProblem}


\begin{homeworkProblem}
	\begin{enumerate}[(a)]
		\item Let $A=\{0^k u 0^k \mid k\geq 1\text{ and } u\in\Sigma^\ast\}.$ Show that $A$ is regular.
		\item Let $B=\{0^k 1u 0^k \mid k\geq 1\text{ and } u\in\Sigma^\ast\}.$ Show that $B$ is not regular.
	\end{enumerate}
\end{homeworkProblem}


\begin{homeworkProblem}
	Let $B$ be the set of all strings with even length that contain at least one $\mathsf{1}$ in their first half.
	\begin{enumerate}[(a)]
		\item Is B regular? Please show your proof.
		\item Let $C = \{w \mid w \in B \text{ or } w \text{ has odd length}\}$. Is C regular? Please show your proof.
		\item Let $D = \{w \mid w \text{ has even length but } w \notin B\}$.  Is D regular? Please show your proof.
	\end{enumerate}
\end{homeworkProblem}


\begin{homeworkProblem}
	\begin{enumerate}[(a)]
		\item Let $A$ be an infinite regular language. Prove that $A$ can be split into two infinite disjoint regular subsets.
		\item Let $B$ and $D$ be two languages. Write 
		$B\subsetneq D$ if $B \subseteq D$ and $D$ contains infinitely many strings that are not in $B$. Show that if $B$ and $D$ are two regular languages where $B\subsetneq D$, then we can find a regular language  $C$ where $B\subsetneq C\subsetneq D$.
	\end{enumerate}
\end{homeworkProblem}
